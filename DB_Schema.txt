CREATE TABLE students (
    id SERIAL PRIMARY KEY,
    roll_no INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    subject VARCHAR(100) NOT NULL,
    section VARCHAR(10) NOT NULL
);
