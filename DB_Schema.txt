    id SERIAL PRIMARY KEY,
    roll_no INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    subject VARCHAR(100) NOT NULL,
    section VARCHAR(10) NOT NULL,
    grade INT,
);

CREATE TABLE teacher_schedule (
    id SERIAL PRIMARY KEY,
    day VARCHAR(15) NOT NULL,
    section VARCHAR(10) NOT NULL,
    room VARCHAR(10),
    time_slot VARCHAR(20) NOT NULL,
    subject VARCHAR(50) NOT NULL,
    grade INT
);


ALTER TABLE teacher_schedule
ADD CONSTRAINT unique_teacher_schedule
UNIQUE (day, section, time_slot, subject);

-----------------------------------------

CREATE TABLE assignments (
    id SERIAL PRIMARY KEY,
    exercise VARCHAR(100) NOT NULL,
    subject VARCHAR(100) NOT NULL,
    section VARCHAR(10) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-----------------------------------------


CREATE TABLE assignment_results (
    id SERIAL PRIMARY KEY,

    assignment_id INT NOT NULL,
    student_id INT NOT NULL,

    marks INT CHECK (marks BETWEEN 1 AND 5),
    remarks TEXT,

    FOREIGN KEY (assignment_id)
        REFERENCES assignments(id)
        ON DELETE CASCADE,

    FOREIGN KEY (student_id)
        REFERENCES students(id)
        ON DELETE CASCADE,

    UNIQUE (assignment_id, student_id)
);




-----------------------------------------



CREATE TABLE class_remarks (
    id SERIAL PRIMARY KEY,

    schedule_id INTEGER NOT NULL
        REFERENCES teacher_schedule(id),

    class_date DATE NOT NULL,

    topics_covered TEXT NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE(schedule_id, class_date)
);



-----------------------------------------

CREATE TABLE student_remarks (
    id SERIAL PRIMARY KEY,

    schedule_id INTEGER NOT NULL
        REFERENCES teacher_schedule(id),

    class_date DATE NOT NULL,

    student_id INTEGER NOT NULL
        REFERENCES students(id),

    remarks TEXT,

    distracted BOOLEAN DEFAULT FALSE,
    participating BOOLEAN DEFAULT FALSE,
    great_effort BOOLEAN DEFAULT FALSE,
    class_disturbance BOOLEAN DEFAULT FALSE,
    late_arrival BOOLEAN DEFAULT FALSE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE(schedule_id, class_date, student_id)
);
